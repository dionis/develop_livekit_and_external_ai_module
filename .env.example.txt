# --- ARTalk Installation -------------------------------------------------
# Absolute path to the cloned ARTalk repository.
# Set automatically by setup.sh.
ARTALK_PATH=/path/to/external_models/ARTalk


# --- Model Load Strategy -------------------------------------------------
# Controls HOW the ARTalk model pipeline is initialised at startup.
#
#   from_scratch   (DEFAULT)
#       Loads each component individually from checkpoint files:
#         - BitwiseARModel  ← assets/ARTalk_wav2vec.pt  (or ARTalk_mimi.pt)
#         - GAGAvatar       ← assets/GAGAvatar/GAGAvatar.pt
#         - FLAME           ← assets/FLAME_with_eye.pt
#       Gives maximum flexibility; suitable for production and custom avatars.
#
#   example_models
#       Delegates ALL loading to ARTalk's own ARTAvatarInferEngine class,
#       which is the exact class used by the official Gradio demo (inference.py).
#       Fastest way to get to a working demo without customisation.
#       The watermark monkey-patch is applied automatically.
#
ARTALK_MODEL_STRATEGY=from_scratch
# ARTALK_MODEL_STRATEGY=example_models


# --- Avatar Appearance (Shape ID) ----------------------------------------
# Selects which 3D avatar face to render.
#
# Built-in options (no preprocessing required):
#
#   mesh          (DEFAULT)
#       Neutral FLAME 3D mesh — no texture, no identity.
#       Fastest and most lightweight; good for testing audio sync.
#
#   11.jpg
#       Pre-tracked GAGAvatar identity bundled with ARTalk (assets/GAGAvatar/tracked.pt).
#       Female face used in the official Gradio demo (Chinese audio examples).
#
#   12.jpg
#       Pre-tracked GAGAvatar identity bundled with ARTalk (assets/GAGAvatar/tracked.pt).
#       Male face used in the official Gradio demo (English/Japanese/Chinese examples).
#
# Custom avatar (requires preprocessing):
#   /absolute/path/to/your_portrait.jpg
#       Any front-facing portrait image (JPG or PNG).
#       The plugin's image_preprocessor.py will automatically:
#         1. Run GAGAvatar_track to extract FLAME parameters from the portrait
#         2. Store the result in assets/GAGAvatar/tracked.pt under the image basename
#         3. Return the basename (e.g. "alice.jpg") as the shape_id for future runs
#       After the first run, set AVATAR_IMAGE=alice.jpg (basename only) to skip
#       re-tracking on subsequent starts.
#
AVATAR_IMAGE=mesh
# AVATAR_IMAGE=11.jpg
# AVATAR_IMAGE=12.jpg
# AVATAR_IMAGE=/absolute/path/to/my_portrait.jpg


# --- Speaking Style (Style ID) -------------------------------------------
# Controls the motion style / expressiveness of the generated animation.
# All styles are loaded from assets/style_motion/<style_id>.pt
# Downloaded automatically by build_resources.sh / setup.sh.
#
#   default       — No style condition; the model uses its learned prior.
#
#   Neutral / Natural moods (most stable for long speech):
#     natural_0   — Calm, relaxed delivery          (most neutral)
#     natural_1   — Slightly expressive natural
#     natural_2   — Moderate expressiveness
#     natural_3   — Lively natural
#     natural_4   — More dynamic natural
#     natural_5   — Energetic natural
#     natural_6   — Very dynamic natural
#     natural_7   — Most expressive natural variant
#
#   Positive moods:
#     happy_0     — Lightly happy
#     happy_1     — Moderately happy
#     happy_2     — Very happy
#
#   Curious / Questioning:
#     curious_0   — Curious, questioning tone
#
#   Doubtful / Uncertain:
#     doubtful_0  — Mildly doubtful
#     doubtful_1  — Strongly doubtful
#
#   Negative moods:
#     angry_0     — Angry, intense delivery
#
ARTALK_STYLE_ID=natural_0
# ARTALK_STYLE_ID=default
# ARTALK_STYLE_ID=happy_0
# ARTALK_STYLE_ID=curious_0
# ARTALK_STYLE_ID=doubtful_0
# ARTALK_STYLE_ID=angry_0


# --- Audio Encoder -------------------------------------------------------
# ARTalk ships with two audio encoder backends.
# The encoder is auto-detected from whichever checkpoint file is present in assets/;
# you do NOT normally need to set this manually.
#
#   wav2vec  (DEFAULT — recommended)
#       Uses Facebook's wav2vec 2.0 features.
#       Checkpoint: assets/ARTalk_wav2vec.pt
#       Best lip-sync accuracy; used in all official demos.
#
#   mimi
#       Uses the Mimi neural audio codec as the encoder backbone.
#       Checkpoint: assets/ARTalk_mimi.pt  (not downloaded by build_resources.sh by default)
#       Alternative encoder; may offer different trade-offs.
#
# To use mimi: download ARTalk_mimi.pt manually from HuggingFace and place in assets/.
# ARTALK_AUDIO_ENCODER=wav2vec  # (auto-detected; do not set unless you know why)

LIVEKIT_URL=wss://your-livekit-server.livekit.cloud
LIVEKIT_API_KEY=your_livekit_api_key
LIVEKIT_API_SECRET=your_livekit_api_secret
